134/* NODE VARIABLES
135 *
136 * For node variables, the application code will define a structure of the form:
137 *
138 * Typedef struct {
139 * ...
140 * } ModuleNvDefs
141 *
142 * Where ModuleNvDefs is a structure type that defines the meaning and structure
143 * of each node variable as used by that application.
144 *
145 * NzVPtr is then defined to point to the NodeBytes member of the union, thus giving the
146 * generic FLiM code access to the node variables table without requiring any knowledge
147 * of the structure of the node variables themselves.
148 */
149typedef BYTE NodeBytes[];
150typedef union {
151 ModuleNvDefs moduleNVs;
152 NodeBytes nodevars; // Do not change this as it is used by FLiM.c
153} NodeVarTable;

171// State machine for FliM operations
172
173enum FLiMStates
174{
175 fsSLiM=0, // In SLiM, button not pressed
176 fsFLiM, // In FLiM mode
177 fsPressed, // Button pressed, waiting for long enough
178 fsFlashing, // Button pressed long enough,flashing now
179 fsFLiMSetup, // In FLiM setup mode (LED flashing)
180 fsFLiMRelease, // Release FLiM back to SLiM
181 fsSetupDone, // Exiting setup mode
182 fsFLiMLearn, // In FLiM learn mode
183 fsPressedFLiM, // Pressed whilst in FLiM mode
184 fsPressedSetup, // Pressed during setup
185#ifdef TEST_MODE
186 fsPressedTest, // Pressed during test
187 fsTestMode, // Self test mode
188 fsNextTest, // Move on to next test
189 fsTestInput, // Input to current test mode
190#endif
191 fsNoData = 0xFF // Uninitialised memory indicates no FLiM data present
192};

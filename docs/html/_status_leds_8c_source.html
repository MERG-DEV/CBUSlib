62/**
63 * Initialise with both LEDs off.
64 */
65void initStatusLeds(void)
66{
67 TRIS_LED1Y = 0;
68 TRIS_LED2G = 0;
69 LED1Y = LED_OFF;
70 LED2G = LED_OFF;
71 flashState = flNone;
72 flickerTime.Val = 0;
73}
74/**
75 * Set the Green LED and turn of the Yellow LED.
76 */
77void setSLiMLed(void)

79 LED1Y = 0;
80 LED2G = 1;
81 flashState = flNone;
82}
83/**
84 * Set the yellow LED and turn off the green LED.

93/**
94 * Sets either the FLiM or the SLiM LED.

105/**

114/**
115 * Start the FLiM LED flashing at the specified rate.

124/**
125 * Process the FLiM LED flashing. Call regularly.
126 */
127void checkFlashing(void)
128{
129 switch (flashState)
130 {
131 case flFLiMSlow:
132 if (tickTimeSince(flashTime) > SLOW_FLASH_TIME)
133 doFLiMFlash();
134 break;
135
136 case flFLiMFast:
137 if (tickTimeSince(flashTime) > FAST_FLASH_TIME)
138 doFLiMFlash();
139 break;
140 }
141 switch(flickerState) {
142 case flickWaitingOn:
143 // turn on the LED - FLiM this would be the green SLiM led
144 if (flimState == fsSLiM) {
145 LED1Y = 1;
146 } else {
147 LED2G = 1;
148 }
149 flickerState = flickOn;
150 break;
151 case flickOn:
152 // time to turn off yet?
153 if (tickGet() >= flickerTime.Val) {
154 // turn off the LED - in FLiM this would be the green SLiM led
155 if (flimState == fsSLiM) {
156 LED1Y = 0;
157 } else {
158 LED2G = 0;
159 }
160 flickerState = flickOff;
161 }
162 break;
163 }
164}

169void shortFlicker() {

171 flickerTime.Val = tickGet() + SHORT_FLICKER_TIME;

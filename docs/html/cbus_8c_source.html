168// Send CBUS event with optional data bytes - works out appropriate opcode
169BOOL cbusSendEventWithData( BYTE cbusNum, WORD eventNode, WORD eventNum, BOOL onEvent, BYTE *msg, BYTE datalen )
170{
171 BOOL ret;
172 msg[d0] = OPC_ACON; // Start with long event opcode
173
174 if (eventNode == 0)
175 {
176 msg[d0] |= 0x08; // Short event opcode
177 eventNode = nodeID; // Add module node id for diagnostics
178 }
179
180 if (!onEvent)
181 msg[d0] |= 0x01; // Off event
182
183 if (datalen > 0)
184 msg[d0] |= (datalen << 5); // Opcode for event + data
185
186 msg[d3] = eventNum>>8;
187 msg[d4] = eventNum & 0xFF;
188
189 ret = cbusSendMsgNN( cbusNum, eventNode, msg );
190
191 #if defined(CBUS_OVER_CAN)
192 if ((cbusNum == CBUS_OVER_CAN) || (cbusNum == 0xFF) )
193 canQueueRx( (CanPacket*)msg ); // Queue event into receive buffer so module can be taught its own events
194
195 #endif
196 return ret;
197} // cbusSendEventWithData

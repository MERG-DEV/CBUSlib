78/*********************************************************************
79* Function: void InitTicker()
80*
81* PreCondition: none
82*
83* Input: none
84*
85* Output: none
86*
87* Side Effects: TMR0 for PIC18 is configured for calculating
88* the correct symbol times. TMR2/3 for PIC24/dsPIC
89* is configured for calculating the correct symbol
90* times
91*
92* Overview: This function will configure the UART for use at
93* in 8 bits, 1 stop, no flowcontrol mode
94*
95* Note: The timer interrupt is enabled causing the timer
96* roll over calculations. Interrupts are required
97* to be enabled in order to extend the timer to
98* 4 bytes in PIC18. PIC24/dsPIC version do not
99* enable or require interrupts
100********************************************************************/
101
102void initTicker(unsigned char priority)
103{
104 BYTE divider, i;
105
106 setclkMHz();
107 divider = 0;
108
109 for (i=clkMHz;i>0;i>>=1) // Work out timer prescaler value from clock MHz
110 divider++;
111
112#if defined(__18CXX) || defined (__XC8__)
113 TMR_CON = 0b00000000 | divider; // Enable internal clock, prescaler on and set prescaler value
114 TMR_IP = priority;
115 TMR_IF = 0;
116 TMR_IE = 1;
117 TMR_ON = 1;
118
119 timerExtension1 = 0;
120 timerExtension2 = 0;
121#elif defined(__dsPIC30F__) || defined(__dsPIC33F__) || defined(__PIC24F__) || defined(__PIC24FK__) || defined(__PIC24H__)
122 T2CON = 0b0000000000001000 | CLOCK_DIVIDER_SETTING;
123 T2CONbits.TON = 1;
124#elif defined(__PIC32MX__)
125 CloseTimer2();
126 WriteTimer2(0x00);
127 WriteTimer3(0x00);
128 WritePeriod3(0xFFFF);
129 OpenTimer2((T2_ON|T2_32BIT_MODE_ON|CLOCK_DIVIDER_SETTING),0xFFFFFFFF);
130#else
131 #error " timer implementation required for stack usage."
132#endif
133}

136/*********************************************************************
137* Function: void tickGet()
138*
139* PreCondition: none
140*
141* Input: none
142*
143* Output: TickValue - the current tick time
144*
145* Side Effects: PIC18 only: the timer interrupt is disabled
146* for several instruction cycles while the
147* timer value is grabbed. This is to prevent a
148* rollover from incrementing the timer extenders
149* during the read of their values
150*
151* Overview: This function returns the current time
152*
153* Note: PIC18 only:
154* caution: this function should never be called
155* when interrupts are disabled if interrupts are
156* disabled when this is called then the timer
157* might rollover and the byte extension would not
158* get updated.
159********************************************************************/
160DWORD tickGet(void)
161{
162 TickValue currentTime;
163
164#if defined(__18CXX) || defined(__XC8__)
165 //BYTE failureCounter;
166 BYTE IntFlag1;
167 BYTE IntFlag2;
168
169 /* copy the byte extension */
170 currentTime.byte.b2 = 0;
171 currentTime.byte.b3 = 0;
172 /* disable the timer to prevent roll over of the lower 16 bits while before/after reading of the extension */
173 TMR_IE = 0;
174#if 1
175 do
176 {
177 IntFlag1 = TMR_IF;
178 currentTime.byte.b0 = TMR_L;
179 currentTime.byte.b1 = TMR_H;
180 IntFlag2 = TMR_IF;
181 } while(IntFlag1 != IntFlag2);
182
183 if( IntFlag1 > 0 )
184 {
185 TMR_IF = 0;
186 timerExtension1++;
187 if(timerExtension1 == 0)
188 {
189 timerExtension2++;
190 }
191 }
192#else
193
194
195 failureCounter = 0;
196 /* read the timer value */
197 do
198 {
199 currentTime.byte.b0 = TMR_L;
200 currentTime.byte.b1 = TMR_H;
201 } while( currentTime.word.w0 == 0xFFFF && failureCounter++ < 3 );
202
203 //if an interrupt occurred after IE = 0, then we need to figure out if it was
204 //before or after we read TMR0L
205 if(TMR_IF)
206 {
207 //if(currentTime.byte.b0<10)
208 {
209 //if we read TMR0L after the rollover that caused the interrupt flag then we need
210 //to increment the 3rd byte
211 currentTime.byte.b2++; //increment the upper most
212 if(timerExtension1 == 0xFF)
213 {
214 currentTime.byte.b3++;
215 }
216 }
217 }
218#endif
219
220 /* copy the byte extension */
221 currentTime.byte.b2 += timerExtension1;
222 currentTime.byte.b3 += timerExtension2;
223
224 /* enable the timer*/
225 TMR_IE = 1;
226
227#elif defined(__dsPIC30F__) || defined(__dsPIC33F__) || defined(__PIC24F__) || defined(__PIC24FK__) || defined(__PIC24H__) || defined(__PIC32MX__)
228 currentTime.word.w1 = TMR3;
229 currentTime.word.w0 = TMR2;
230 if( currentTime.word.w1 != TMR3 )
231 {
232 currentTime.word.w1 = TMR3;
233 currentTime.word.w0 = TMR2;
234 }
235#else
236 #error "Symbol timer implementation required for stack usage."
237#endif
238 return currentTime.Val;
239} // tickGet

52// CBUS EEPROM usage is at the top of EEPROM, which leaves space from 0 upwards for use by the MLA.
53// Note that if you are using internal EEPROM for MLA, you must reduce the amount of EEPROM declared to MLA by the amount used here by CBUS
54
55#ifdef EE256
56 #define EE_TOP 0xFF
57 #define EE_BOTTOM 0x00
58 #define SET_EADDRH(val) // EEPROM high address reg not present in 2480/2580, so just do nothing
59#endif
60
61#ifdef EE1024
62 #define EE_TOP 0x3FF
63 #define EE_BOTTOM 0x00
64 #define SET_EADDRH(val) EEADRH = val // EEPROM high address is present, so write value
65#endif
66
67// Top bytes for Bootloader and CBUS library usage - these values are address offsets into EEPROM
68// Library uses top 8 for general storage then next 32 for event status
69// application starts at EE_TOP-40
70
71#define EE_BOOT_FLAG ((BYTE*)(EE_TOP)) // Set to FF to enter bootloader
72#define EE_CAN_ID ((BYTE*)(EE_TOP-1)) // 7 bit CANID 1 to 127
73#define EE_NODE_ID ((WORD*)(EE_TOP-3)) // 16 bit value for node number. Size = 2
74#define EE_FLIM_MODE ((BYTE*)(EE_TOP-4)) // Enumerated value for SLiM/FLiM mode
75#define EE_VERSION ((BYTE*)(EE_TOP-5)) // Indicates if EEPROM needs initialising or upgrading
76#ifdef AREQ_SUPPORT
77#define EE_AREQ_STATUS ((BYTE*)(EE_TOP-39)) // Event status bits. Size=34
78#define EE_APPLICATION ((BYTE*)(EE_TOP-40))
79#else
80#define EE_APPLICATION ((BYTE*)(EE_TOP-40)) // this could be moved but better for compatability if stays in same location
81#endif
82
83#endif // __EEPROM_H

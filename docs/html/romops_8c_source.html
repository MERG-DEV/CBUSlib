96/**
97 * Flash block write. flash 64 byte buffer.
98 * Fast write, this requires no 0 to 1 bit changes, only 1 to 0 bits are allowed.
99 * Or use write_flash_long with erase before write
100 */
101 void writeFlashShort(void)
102 {
103
104#ifdef __XC8__
105 TBLPTR = flashblock & ~(64 - 1); //force row boundary
106 di(); // disable all interrupts ERRATA says this is needed before TBLWT
107 for (unsigned char i=0; i<64; i++) {
108 TABLAT = flashbuf[i];
109 asm("TBLWT*+");
110 }
111 // Note from data sheet:
112 // Before setting the WR bit, the Table
113 // Pointer address needs to be within the
114 // intended address range of the 64 bytes in
115 // the holding register.
116 // So we put it back into the block here
117 TBLPTR = flashblock & ~(64 - 1);
118 EECON1bits.EEPGD = 1; // 1=Program memory, 0=EEPROM
119 EECON1bits.CFGS = 0; // 0=ProgramMemory/EEPROM, 1=ConfigBits
120 EECON1bits.FREE = 0; // No erase
121 EECON1bits.WREN = 1; // enable write to memory
122
123 EECON2 = 0x55;
124 EECON2 = 0xAA;
125 EECON1bits.WR = TRUE;
126 ei(); // enable all interrupts
127 EECON1bits.WREN = FALSE;
128#else
129 WORD ptr;
130#ifndef CPUF18K
131 BYTE fwCounter;
132#endif
133
134 ptr= (WORD)flashbuf;
135 // check we are only writing within the persistent data area
136 if ((flashblock < MIN_WRITEABLE_FLASH) || (flashblock > MAX_WRITEABLE_FLASH)) {
137 // should never reach here
138 // assert
139 return;
140 }
141 // Call back into the application to check if now is a good time to write the flash
142 // as the processor will be suspended for up to 2ms.
143 while (! isSuitableTimeToWriteFlash());
144 di(); // disable all interrupts
145 TBLPTR=flashblock;
146 TBLPTRU = 0x00; // just to make sure
147 FSR0=ptr;
148
149#ifdef CPUF18K
150 flashidx=64; // K series processors can write 64 bytes in one operation
151#else
152 for (fwCounter = 1; fwCounter <=2; fwCounter++ ) // 18F processors need two iterations of 32 bytes each
153 {
154 flashidx=32;
155#endif
156
157_asm
158L_w:
159 MOVF POSTINC0,W,0
160 MOVWF TABLAT,0
161 TBLWTPOSTINC
162 DECF flashidx,F,1
163 BNZ L_w
164 TBLRDPOSTDEC //PUT IT BACK IN THE BLOCK
165_endasm
166
167 EECON1bits.EEPGD = 1; // 1=Program memory, 0=EEPROM
168 EECON1bits.CFGS = 0; // 0=ProgramMemory/EEPROM, 1=ConfigBits
169 EECON1bits.FREE = 0; // No erase
170 EECON1bits.WREN = 1; // enable write to memory
171 EECON2 = 0x55; // write 0x55
172 EECON2 = 0xaa; // write 0xaa
173 EECON1bits.WR = 1; // start writing
174 EECON1bits.WREN = 0; // disable write to memory
175_asm
176 TBLRDPOSTINC // Table pointer ready for next 32
177_endasm
178 TBLPTRU = 0x00; // this will be set to 1 after writing 0xFFFF so reset it
179#endif
180
181#ifdef CPUF18F
182 }
183#endif
184 ei(); // enable all interrupts
185
186}

189/**
190 * Flash block write. flash 64 byte buffer with block erase.
191 */
192void writeFlashWithErase(void)
193{
194 // Call back into the application to check if now is a good time to write the flash
195 // as the processor will be suspended for up to 2ms.
196 while (! isSuitableTimeToWriteFlash());
197 // Erase block first
198 TBLPTR=flashblock;
199 TBLPTRU = 0x00; // just to make sure
200 EECON1bits.EEPGD = 1; // 1=Program memory, 0=EEPROM
201 EECON1bits.CFGS = 0; // 0=Program memory/EEPROM, 1=ConfigBits
202 EECON1bits.WREN = 1; // enable write to memory
203 EECON1bits.FREE = 1; // enable row erase operation
204 di(); // disable all interrupts
205 EECON2 = 0x55; // write 0x55
206 EECON2 = 0xaa; // write 0xaa
207 EECON1bits.WR = 1; // start erasing
208 ei(); // enable all interrupts
209 EECON1bits.WREN = 0; // disable write to memory
210 //Now write data to flash
211 writeFlashShort();
212}

229/**
230 * Flash block read. Read flash through a 64 byte buffer with write back management.
231 * valid:3 //must be 101 (5) to be valid
232 * loaded:1 // if buffer is loaded
233 * modified:1 //flag if buffer is modified
234 * zeroto1:1 //flag if long write with block erase
235 */
236 /**
237 * Read a byte from flash. If the required buffer is currently loaded then use the value
238 * stored there since it may have been modified. Otherwise load the buffer from flash
239 * before returning the value.
240 * @param addr the address to be read from Flash
241 * @return the byte read from Flash
242 */
243BYTE readFlashBlock(WORD flashAddr)
244{
245 WORD ptr;
246 // check we are only reading within the persistent data area
247 if ((flashAddr < MIN_WRITEABLE_FLASH) || (flashAddr > MAX_WRITEABLE_FLASH)) {
248 // should never reach here
249 // assert
250 return 0xff;
251 }
252
253 if(flashFlags.valid !=5)
254 {
255 flashFlags.asByte=5; //force reload
256 }
257
258 if(flashFlags.loaded && flashblock!=(flashAddr & 0XFFC0))
259 {
260 //detected access from a different block so we need to write this one (if it has been changed)
261 flushFlashImage();
262 flashFlags.asByte=5;
263 }
264
265 if(!flashFlags.loaded)
266 {
267 // load the buffer
268 flashblock = flashAddr & 0xFFC0;
269#ifdef __XC8__
270 EECON1=0X80; // access to flash
271 TBLPTR = flashblock;
272 for (unsigned char i=0; i<64; i++) {
273 asm("TBLRD*+");
274 flashbuf[i] = TABLAT;
275 }
276#else
277 //load the buffer
278 ptr= (WORD)flashbuf;
279 FSR0=ptr;
280 flashblock = flashAddr & 0XFFC0;
281 TBLPTR=flashblock;
282 TBLPTRU = 0x00; // just to make sure
283 EECON1=0X80;
284 flashidx=64;
285_asm
286// MOVLB FLASHBUFPAGE
287READ_BLOCK:
288 TBLRDPOSTINC
289 NOP
290 MOVFF TABLAT,POSTINC0
291 DECF flashidx,F,1
292 BNZ READ_BLOCK
293_endasm
294#endif
295 TBLPTRU = 0x00; // this will be set to 1 after reading 0xFFFF so reset it
296 flashFlags.loaded = TRUE;
297
298 }
299 return flashbuf[flashAddr & 0X3F];
300}

303/**
304 * Write a byte to the FLASH image. You may need to flush current image to Flash if necessary.
305 * @param addr the destination address of the byte to be written
306 * @param data the data byte to be written
307 */
308void writeFlashImage(BYTE * addr, BYTE data)
309{
310 unsigned char *offset;
311 // check we are only writing within the persistent data area
312 if (((WORD)addr < MIN_WRITEABLE_FLASH) || ((WORD)addr > MAX_WRITEABLE_FLASH)) {
313 // should never reach here
314 // assert
315 return;
316 }
317
318 if(flashFlags.valid !=5)
319 {
320 flashFlags.valid=5; //force reload
321 }
322
323 if (!flashFlags.loaded || flashblock!=((WORD)addr & 0XFFC0))
324 readFlashBlock((WORD)addr);
325
326 offset = &flashbuf[(WORD)addr & 0x3F];
327
328 if(data !=*offset)
329 flashFlags.modified=1;
330
331 if(data & ~*offset)
332 flashFlags.zeroto1=1;
333 *offset=data;
334}

377/**
378 * Read a byte from data EEPROM.
379 * @param addr the address to be read
380 * @return the byte from EEPROM
381 */
382BYTE ee_read(WORD addr) {
383 while (EECON1bits.WR) // Errata says this is required
384 ;
385 // EEADRH = addr >> 8; // High byte of address to read
386 SET_EADDRH(addr >> 8);
387 EEADR = addr & 0xFF; /* Low byte of Data Memory Address to read */
388 EECON1bits.EEPGD = 0; /* Point to DATA memory */
389 EECON1bits.CFGS = 0; /* Access program FLASH/Data EEPROM memory */
390 EECON1bits.RD = 1; /* EEPROM Read */
391 while (EECON1bits.RD)
392 ;
393#ifdef __XC8
394 asm("NOP"); /* data available after a NOP */
395#else
396 _asm
397 nop
398 _endasm
399#endif
400 return EEDATA;
401}

403/**
404 * Write one byte to data EEPROM.
405 * We verify at end and try again if necessary.
406 * @param addr the address to be written
407 * @param data the data to be written
408 */
409void ee_write(WORD addr, BYTE data) {
410 do {
411 SET_EADDRH(addr >> 8); // High byte of address to write
412 EEADR = addr & 0xFF; /* Low byte of Data Memory Address to write */
413 EEDATA = data;
414 EECON1bits.EEPGD = 0; /* Point to DATA memory */
415 EECON1bits.CFGS = 0; /* Access program FLASH/Data EEPROM memory */
416 EECON1bits.WREN = 1; /* Enable writes */
417 di(); /* Disable Interrupts */
418 EECON2 = 0x55;
419 EECON2 = 0xAA;
420 EECON1bits.WR = 1;
421#ifdef __XC8__
422 while (EECON1bits.WR)
423 ;
424 //asm("BTFSC EECON1, 1"); // should wait until WR clears
425#else
426 _asm nop
427 nop _endasm
428#endif
429 ei(); /* Enable Interrupts */
430 while (!EEIF)
431 ;
432 EEIF = 0;
433 EECON1bits.WREN = 0; /* Disable writes */
434 } while (ee_read(addr) != data);
435}

437/**
438 * Read a WORD (16 bit) word from EEPROM.
439 * Data is stored in little endian format
440 * @param addr the address to be read
441 * @return the WORD from EEPROM
442 */
443WORD ee_read_short(WORD addr)
444
445{
446 WORD ee_addr = addr;
447 WORD ret = ee_read(ee_addr++);
448
449 ret = ret | ((WORD)ee_read(ee_addr) << 8);
450 return ret;
451}

454 * Write a WORD (16 bit) data to EEPROM.
455 * Data is stored in little endian format.
456 * @param addr the address to be written

459void ee_write_short(WORD addr, WORD data) {

462 ee_write(ee_addr, (BYTE)(data>>8));

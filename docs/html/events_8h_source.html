54/* EVENTS
55 *
56 * The events are stored in tables in flash (flash is faster to read than EEPROM).
57 * Separate tables are used for Produced events and Consumed events since they require
58 * different data and different lookup schemes.
59 * The Produced events are looked up by the action that caused them.
60 * The Consumed events are looked up by the CBUS event to find the module specific actions
61 * to perform. The lookup is done using a hash table to find the index into the event2Actions table.
62 * The action2Event and event2Actions tables are stored in Flash whilst the hashtable
63 * lookup for the event2Actions table is stored in RAM.
64 *
65 * For Produced events the event is taught using the action stored in the EV field of the
66 * CBUS message.
67 * For Consumed events the actions are taught using the EV field of the CBUS message.
68 * Multiple actions can be specified for a Consumed event. This can be used to set up
69 * routes with a single event.
70 *
71 * The generic FLiM library code handles the teaching (Learn) and unlearning of events
72 * and the storage of the events. The application code just needs to process a consumed
73 * event's actions and to produce actions using the application's actions.
74 *
75 */

104// EVENT DECODING
105// An event opcode has bits 4 and 7 set, bits 1 and 2 clear
106// An ON event opcode also has bit 0 clear
107// An OFF event opcode also has bit 0 set
108//
109// eg:
110// ACON 90 1001 0000
111// ACOF 91 1001 0001
112// ASON 98 1001 1000
113// ASOF 99 1001 1001
114//
115// ACON1 B0 1011 0000
116// ACOF1 B1 1011 0001
117// ASON1 B8 1011 1000
118// ASOF1 B9 1011 1001
119//
120// ACON2 D0 1101 0000
121// ACOF2 D1 1101 0001
122// ASON2 D8 1101 1000
123// ASOF2 D9 1101 1001
124//
125// ACON3 F0 1111 0000
126// ACOF3 F1 1111 0001
127// ASON3 F8 1111 1000
128// ASOF3 F9 1111 1001
129//
130// ON/OFF determined by d0
131// Long/Short determined by d3
132// num data bytes determined by d7,d6
133//
134#define EVENT_SET_MASK 0b10010000
135#define EVENT_CLR_MASK 0b00000110
136#define EVENT_ON_MASK 0b00000001
137#define EVENT_SHORT_MASK 0b00001000

110//*******************************************************************************
111// Initialise CAN
112
113void canInit(BYTE busNum, BYTE initCanID) {
114
115 larbCount = 0;
116 txErrCount = 0;
117 txTimeoutCount = 0;
118 canTransmitFailed = FALSE;
119 canTransmitTimeout.Val = 0;
120 maxCanTxFifo = 0;
121 maxCanRxFifo = 0;
122 rxOflowCount = 0;
123 txOflowCount = 0;
124 txIndexNextFree = 0;
125 txIndexNextUsed = 0;
126 rxIndexNextFree = 0;
127 rxIndexNextUsed = 0;
128 txFifoUsage = 0;
129 rxFifoUsage = 0;
130
131 IPR5 = CAN_INTERRUPT_PRIORITY; // CAN interrupts priority
132
133 // Put module into Configuration mode.
134 CANCON = 0b10000000;
135
136 // Wait for config mode
137 while (CANSTATbits.OPMODE2 == 0);
138
139 /*
140 * The CAN baud rate pre-scaler is preset by the bootloader, so this code is written to be clock speed independent.
141 * The global variable clkMHz is set by reading in CAN prescaler value from BRGCON or CiCFG for use by any other routines
142 * that need to know the clock speed. Note that on the PIC32 the Peripheral bus prescaler setting will also need to be taken
143 * into accounts when configuring time related values for peripherals that use pbclk.
144 */
145
146#ifdef __C32__
147
148 clkMHz = (( CiCFG & 0x3F ) + 1 ) << 2; // Convert CAN config register into clock MHz
149
150#else
151
152 ECANCON = 0b10110000; // ECAN mode 2 with FIFO, FIFOWM = 1 (init when four spaces left), init to first RX buffer
153 BSEL0 = 0; // Use all 8 buffers for receive FIFO, as they do not work as a FIFO for transmit
154
155 #ifdef CPUF18F
156 BRGCON1 = 0B00000011; // ??? Set for 16MHz clock for the moment, will be 64MHz when PLL used, then remove from here when set in bootloader
157 #else
158 BRGCON1 = 0b00001111; // 16MHz resonator + PLL = 64MHz clock
159 #endif
160
161 clkMHz = (( BRGCON1 & 0x3F ) + 1 ) << 2; // Convert BRGCON1 value into clock MHz.
162
163#endif
164
165 /* The CAN bit rates used for CBUS are calculated as follows:
166
167 * Sync segment is fixed at 1 Tq
168 * We are using propogation time of 7tq, phase 1 of 4Tq and phase2 of 4Tq.
169 * Total bit time is Sync + prop + phase 1 + phase 2
170 * or 16 * Tq in our case
171 * So, for 125kbits/s, bit time = 8us, we need Tq = 500ns
172 * To get 500nS, we set the CAN bit rate prescaler, in BRGCON1, to half the FOsc clock rate.
173 * For example, 16MHz oscillator using PLL, Fosc is 64MHz, Tosc is 15.625nS, so we use prescaler of 1:32 to give Tq of 500nS (15.625 x 32)
174 * Having set Tq to 500nS, all other CAN timings are relative to Tq, so do not need changing with processor clock speed
175 */
176
177 // BRGCON values used are as follows, now preset in the bootloader to make this routine clock speed independent:
178
179 //BRGCON1 = 0b00000011; // 4MHz resonator + PLL = 16MHz clock (or 16MHz resonator no PLL)
180 //BRGCON1 = 0b00000111; // 8MHz resonator + PLL = 32MHz clock`
181 //BRGCON1 = 0b00001111; // 16MHz resonator + PLL = 64MHz clock
182
183 BRGCON2 = 0b10011110; // freely programmable, sample once, phase 1 = 4xTq, prop time = 7xTq
184 BRGCON3 = 0b00000011; // Wake-up enabled, wake-up filter not used, phase 2 = 4xTq
185
186 // Continue CAN initialisation from where bootloader left off
187
188 CIOCON = 0b00100000; // TX drives Vdd when recessive, CAN capture to CCP1 disabled
189
190
191
192// Setup masks so all filter bits are ignored apart from EXIDEN
193 RXM0SIDH = 0;
194 RXM0SIDL = 0x08;
195 RXM0EIDH = 0;
196 RXM0EIDL = 0;
197 RXM1SIDH = 0;
198 RXM1SIDL = 0x08;
199 RXM1EIDH = 0;
200 RXM1EIDL = 0;
201
202 // Set filter 0 for standard ID only to reject bootloader messages
203 RXF0SIDL = 0x80;
204
205 // Link all filters to RXB0 - maybe only neccessary to link 1
206 RXFBCON0 = 0;
207 RXFBCON1 = 0;
208 RXFBCON2 = 0;
209 RXFBCON3 = 0;
210 RXFBCON4 = 0;
211 RXFBCON5 = 0;
212 RXFBCON6 = 0;
213 RXFBCON7 = 0;
214
215 // Link all filters to mask 0
216 MSEL0 = 0;
217 MSEL1 = 0;
218 MSEL2 = 0;
219 MSEL3 = 0;
220
221// RXM0SIDL = 0; // Clear all mask registers so we get all CAN messages
222// RXM0SIDH = 0; //
223// RXM1SIDL = 0;
224// RXM1SIDH = 0;
225//
226// RXM0EIDL = 0;
227// RXM0EIDH = 0;
228// RXM1EIDL = 0;
229// RXM1EIDH = 0;
230//
231// RXFCON0 = 0; // Disable filters
232// RXFCON1 = 0;
233
234 // Configure the buffers to receive all valid std length messages
235 // RXB0CON = RXB1CON = 0x20; B0CON = B1CON = B2CON = B3CON = B4CON = B5CON = 0;
236
237 // Clear RXFUL bits
238 RXB0CON = 0x20; // IH changed from 0 so that RXM = 01. Receive only std messages
239 RXB1CON = 0x20; // IH changed from 0 so that RXM = 01. Receive only std messages
240 B0CON = 0;
241 B1CON = 0;
242 B2CON = 0;
243 B3CON = 0;
244 B4CON = 0;
245 B5CON = 0;
246
247 BIE0 = 0; // No Rx buffer interrupts (but we do use the high water mark interrupt)
248 TXBIEbits.TXB0IE = 1; // Tx buffer interrupts from buffer 0 only
249 TXBIEbits.TXB1IE = 0;
250 TXBIEbits.TXB2IE = 0;
251 CANCON = 0; // Set normal operation mode
252
253 if (initCanID == 0)
254 {
255 canID = ee_read( (WORD)EE_CAN_ID );
256
257 if (canID == 0xFF)
258 canID = DEFAULT_CANID;
259 }
260 else // use value passed to this routine
261 {
262 canID = initCanID;
263 ee_write((WORD)EE_CAN_ID, canID);
264 }
265
266 // Preload TXB0 with parameters ready for sending CBUS data packets
267
268 TXB0CON = 0;
269 TXB0CONbits.TXPRI0 = 0; // Set buffer priority, so will be sent after any self enumeration packets
270 TXB0CONbits.TXPRI1 = 0;
271 TXB0DLC = 0; // Not RTR, payload length will be set by transmit routine
272 TXB0SIDH = 0b10110000 | ((canID & 0x78) >>3); // Set CAN priority and ms 4 bits of can id
273 TXB0SIDL = (canID & 0x07) << 5; // LS 3 bits of can id and extended id to zero
274
275 // Preload TXB1 with RTR frame to initiate self enumeration when required
276
277 TXB1CON = 0;
278 TXB1CONbits.TXPRI0 = 0; // Set buffer priority, so will be sent before any CBUS data packets but after any enumeration replies
279 TXB1CONbits.TXPRI1 = 1;
280 TXB1DLC = 0x40; // RTR packet with zero payload
281 TXB1SIDH = 0b10110000 | ((canID & 0x78) >>3); // Set CAN priority and ms 4 bits of can id
282 TXB1SIDL = TXB0SIDL; // LS 3 bits of can id and extended id to zero
283
284 // Preload TXB2 with a zero length packet containing CANID for use in self enumeration
285
286 TXB2CON = 0;
287 TXB2CONbits.TXPRI0 = 1; // Set high buffer priority, so will be sent before any CBUS packets
288 TXB2CONbits.TXPRI1 = 1;
289 TXB2DLC = 0; // Not RTR, zero payload
290 TXB2SIDH = 0b10110000 | ((canID & 0x78) >>3); // Set CAN priority and ms 8 bits of can id
291 TXB2SIDL = TXB0SIDL; // LS 3 bits of can id and extended id to zero
292
293 // Initialise enumeration control variables
294
295 enumerationRequired = enumerationInProgress = FALSE;
296 enumerationStartTime.Val = tickGet();
297
298 // Initialisation complete, enable CAN interrupts
299
300 FIFOWMIE = 1; // Enable Fifo 1 space left interrupt
301 ERRIE = 1; // Enable error interrupts
302
303}

305// Set a new can id
306
307BOOL setNewCanId( BYTE newCanId )
308
309{
310 if ((newCanId >= 1) && (newCanId <= 99)) {
311 canID = newCanId;
312 TXB0SIDH &= 0b11110000; // Clear canid bits
313 TXB0SIDH |= ((newCanId & 0x78) >>3); // Set new can id for CUBS packet transmissions
314 TXB0SIDL = ( newCanId & 0x07) << 5;
315
316 TXB1SIDH &= 0b11110000; // Clear canid bits
317 TXB1SIDH |= ((newCanId & 0x78) >>3); // Set new can id for self enumeration frame transmission
318 TXB1SIDL = TXB0SIDL;
319
320 TXB2SIDH &= 0b11110000; // Clear canid bits
321 TXB2SIDH |= ((newCanId & 0x78) >>3); // Set new can id for self enumeration frame transmission
322 TXB2SIDL = TXB0SIDL;
323
324 ee_write((WORD)EE_CAN_ID, newCanId ); // Update saved value
325 return TRUE;
326 } else {
327 return FALSE;
328 }
329}

348// Transmit a packet - DLC must be set to packet length but other fields are set by this routine
349
350BOOL canTX( CanPacket *msg )
351{
352 BYTE* ptr;
353 BOOL fullUp;
354 BYTE hiIndex;
355
356 msg->buffer[con] = 0;
357 msg->buffer[dlc] &= 0x0F; // Ensure not RTR
358 msg->buffer[sidh] = 0b10110000 | ((canID & 0x78) >>3);
359 msg->buffer[sidl] = (canID & 0x07) << 5;
360
361 if (msg->buffer[dlc] > 8)
362 msg->buffer[dlc] = 8;
363
364
365 TXBnIE = 0; // Disable transmit buffer interrupt whilst we fiddle with registers and fifo
366
367 // On chip Transmit buffers do not work as a FIFO, so use just one buffer and implement a software fifo
368
369 if (((txIndexNextUsed == txIndexNextFree) || canTransmitFailed) && (!TXB0CONbits.TXREQ)) // check if software fifo empty and transmit buffer ready
370 {
371 ptr = (BYTE*) & TXB0CON;
372 memcpy(ptr, (void *) msg->buffer, msg->buffer[dlc] + 6);
373
374 larbRetryCount = LARB_RETRIES;
375 canTransmitTimeout.Val = tickGet();
376 canTransmitFailed = FALSE;
377
378 TXB0CONbits.TXREQ = 1; // Initiate transmission
379 fullUp = FALSE;
380 }
381 else // load it into software fifo
382 {
383 if (!(fullUp = (txIndexNextFree == 0xFF)))
384 {
385 memcpy( canTxFifo[txIndexNextFree].buffer, msg->buffer, msg->buffer[dlc] + 6);
386
387 if (++txIndexNextFree == CANTX_FIFO_LEN )
388 txIndexNextFree = 0;
389
390 if (txIndexNextUsed == txIndexNextFree) // check if fifo now full
391 txIndexNextFree = 0xFF; // mark as full
392 }
393 else
394 txOflowCount++;
395
396 // Track buffer usage
397
398 txFifoUsage++;
399 hiIndex = ( txIndexNextFree < txIndexNextUsed ? txIndexNextFree + CANTX_FIFO_LEN : txIndexNextFree);
400 if ((hiIndex - txIndexNextUsed) > maxCanTxFifo )
401 maxCanTxFifo = hiIndex - txIndexNextUsed;
402 }
403
404 TXBnIE = 1; // Enable transmit buffer interrupt
405
406 return !fullUp; // Return true for successfully submitted for transmission
407}

410// Queue a packet into the receive buffer
411// This is used to queue outgoing events back into the rx buffer so that the module
412// can be taught its own events
413// Note that message length must already be set in dlc byte - this is done my cansend which
414// should be called to transmit the message before canQueueRx is called
415
416BOOL canQueueRx( CanPacket *msg )
417
418{
419 FIFOWMIE = 0; // Disable high water mark interrupt so nothing will fiddle with FIFO
420 insertIntoRxFifo( msg );
421 FIFOWMIE = 1;
422 return TRUE;
423}

426// Called by ISR to handle tx buffer interrupt
427
428void checkTxFifo( void )
429{
430 BYTE* ptr;
431
432 canTransmitFailed = FALSE;
433 TXBnIF = 0; // reset the interrupt flag
434
435 if (!TXB0CONbits.TXREQ)
436 {
437 canTransmitTimeout.Val = 0;
438
439 if (txIndexNextUsed != txIndexNextFree) // If data waiting in software fifo, and buffer ready
440 {
441 ptr = (BYTE*) & TXB0CON; // Dest is CAN transmit buffer
442 memcpy(ptr, canTxFifo[txIndexNextUsed].buffer, canTxFifo[txIndexNextUsed].buffer[dlc] + 6);
443 txFifoUsage--;
444
445 larbRetryCount = LARB_RETRIES;
446 canTransmitTimeout.Val = tickGet();
447 canTransmitFailed = FALSE;
448
449 TXB0CONbits.TXREQ = 1; // Initiate transmission
450
451 if (txIndexNextFree == 0xFF)
452 txIndexNextFree = txIndexNextUsed; // clear full status
453
454 if (++txIndexNextUsed == CANTX_FIFO_LEN )
455 txIndexNextUsed = 0;
456
457 TXBnIE = 1; // enable transmit buffer interrupt
458 }
459 else
460 {
461 TXB0CON = 0;
462 TXBnIF = 0;
463 TXBnIE = 0;
464 }
465 }
466 else
467 {
468 TXBnIF = 0;
469 TXBnIE = 1;
470 }
471
472
473
474} // checkTxFifo

476// Called by ISR regularly to check for timeout
477
478void checkCANTimeout( void )
479{
480 if (canTransmitTimeout.Val != 0)
481 if (tickTimeSince(canTransmitTimeout) > CAN_TX_TIMEOUT)
482 {
483 canTransmitFailed = TRUE;
484 txTimeoutCount++;
485 TXB0CONbits.TXREQ = 0; // abort timed out packet
486 checkTxFifo(); // See if another packet is waiting to be sent
487 }
488}

491//*******************************************************************************
492// Called by main loop to check for cbus messages received
493// *msg points to a message buffer where the next message is placed
494// Returns TRUE if a message was found
495
496BOOL canbusRecv(CanPacket *msg)
497{
498
499 CanPacket *ptr;
500 BOOL msgFound;
501
502 FIFOWMIE = 0; // Disable high watermark interrupt so ISR cannot fiddle with FIFOs or enumeration map
503
504 processEnumeration(); // Start or finish canid enumeration if required
505
506
507 // Check for any messages in the software fifo, which the ISR will have filled if there has been a high watermark interrupt
508
509 if (rxIndexNextUsed != rxIndexNextFree)
510 {
511 memcpy(msg->buffer, canRxFifo[rxIndexNextUsed].buffer, canRxFifo[rxIndexNextUsed].buffer[dlc] + 6);
512 rxFifoUsage--;
513
514 if (++rxIndexNextUsed >= CANRX_FIFO_LEN)
515 {
516 rxIndexNextUsed = 0;
517 }
518 msgFound = TRUE;
519 }
520 else // Nothing in software FIFO, so now check for message in hardware FIFO
521 {
522 msgFound = FALSE;
523
524// while (!msgFound && COMSTATbits.NOT_FIFOEMPTY)
525 if (COMSTATbits.NOT_FIFOEMPTY)
526 {
527 ptr = (CanPacket*) _PointBuffer(CANCON & 0x07);
528 RXBnIF = 0;
529// if (COMSTATbits.RXBnOVFL) {
530// maxcanq++; // Buffer Overflow
531// led3timer = 5;
532// LED3 = LED_OFF;
533// COMSTATbits.RXBnOVFL = 0;
534// }
535
536 // Check incoming Canid and initiate self enumeration if it is the same as our own
537
538 if (msgFound = checkIncomingPacket(ptr))
539 memcpy(msg->buffer, (void*) ptr, ptr->buffer[dlc] + 6); // Get message for processing
540
541 // Record and Clear any previous invalid message bit flag.
542 if (IRXIF) {
543 IRXIF = 0;
544 }
545 // Mark that this buffer is read and empty.
546 ptr->buffer[con] &= 0x7f; // clear RXFUL bit
547
548 //TODO - LED flickering to show CBUS activity level
549 // led1timer = 2;
550 // LED1 = LED_ON;
551 } // while !msgFound
552 }
553
554 FIFOWMIE = 1; // Re-enable FIFO interrupts now out of critical section
555 return msgFound;
556}

558// **************************************************************************
559// Insert a CAN packet into the next free location of the receive FIFO
560
561BOOL insertIntoRxFifo( CanPacket *ptr )
562
563{
564 memcpy(canRxFifo[rxIndexNextFree].buffer, ptr, ptr->buffer[dlc] + 6);
565 rxFifoUsage++;
566
567 if (++rxIndexNextFree >= CANRX_FIFO_LEN)
568 {
569 rxIndexNextFree = 0;
570 }
571
572 if (rxIndexNextFree == rxIndexNextUsed)
573 {
574 rxOflowCount++; // Buffer Overflow
575 if (rxIndexNextFree == 0)
576 rxIndexNextFree = CANRX_FIFO_LEN - 1; // On overflow, received packets overwrite last received packet
577 else
578 rxIndexNextFree--;
579
580 }
581 return TRUE;
582} // Insert into RX FIFO

591 CanPacket *ptr;

593
594

610

613 }
614
615 // Record and Clear any previous invalid message bit flag.
616 if (IRXIF)

618
619 // Mark that this buffer is read and empty.
620 ptr->buffer[con] &= 0x7f;

626 if ((hiIndex - rxIndexNextUsed) > maxCanRxFifo )
627 maxCanRxFifo = hiIndex - rxIndexNextUsed;
628
629 } // While hardware FIFO not empty
630 FIFOWMIF = 0;
631} // canFillRxFifo

635void doEnum(BOOL sendResult) {
636 resultRequired = sendResult;
637 if (! enumerationInProgress) {
638 enumerationRequired = TRUE;

641//***********************************************************************************************
642// Check if enumeration pending, if so kick it off providing hold off time has expired
643// If enumeration complete, find and set new can id
644
645void processEnumeration(void)
646
647{
648 BYTE i, newCanId, enumResult;
649
650 if (enumerationRequired && (tickTimeSince(enumerationStartTime) > ENUMERATION_HOLDOFF ))
651 {
652 for (i=1; i< ENUM_ARRAY_SIZE; i++)
653 enumerationResults[i] = 0;
654 enumerationResults[0] = 1; // Don't allocate canid 0
655
656 enumerationInProgress = TRUE;
657 enumerationRequired = FALSE;
658 enumerationStartTime.Val = tickGet();
659 TXB1CONbits.TXREQ = 1; // Send RTR frame to initiate self enumeration
660 }
661 else if (enumerationInProgress && (tickTimeSince(enumerationStartTime) > ENUMERATION_TIMEOUT ))
662 {
663 // Enumeration complete, find first free canid
664
665 // Find byte in array with first free flag. Skip over 0xFF bytes
666 for (i=0; (enumerationResults[i] == 0xFF) && (i < ENUM_ARRAY_SIZE); i++) {
667 ;
668 }
669
670 if ((enumResult = enumerationResults[i]) != 0xFF)
671 {
672 for (newCanId = i*8; (enumResult & 0x01); newCanId++) {
673 enumResult >>= 1;
674 }
675 if ((newCanId >= 1) && (newCanId <= 99)) {
676 canID = newCanId;
677 setNewCanId(canID);
678 if (resultRequired) {
679 cbusSendOpcMyNN( 0, OPC_NNACK, cbusMsg ); // this will get sent for all successful self enums but maybe only required for the ENUM command
680 }
681 }
682 }
683 else
684 {
685 if (resultRequired) {
686 doError(CMDERR_INVALID_EVENT); // seems a strange error code but that's what the spec says...
687 }
688 }
689 enumerationRequired = enumerationInProgress = FALSE;
690
691 }
692} // Process enumeration

695//*******************************************************************************
696// Check incoming packet for RTR enumeration request, any canid clash and zero payload
697// Fills in bitmap with canid if self enumeration in progress
698// Returns TRUE if packet is to be processed as a CBUS message
699
700BOOL checkIncomingPacket(CanPacket *ptr)
701
702{
703 BYTE incomingCanId;
704 BOOL msgFound;
705
706 msgFound = FALSE;
707 incomingCanId = ((ptr->buffer[sidh] << 3) + (ptr->buffer[sidl] >> 5)) & 0x7f;
708
709 if (enumerationInProgress) {
710 arraySetBit( enumerationResults, incomingCanId);
711 } else if (!enumerationRequired && (incomingCanId == canID))
712 {
713 // If we receive a packet with our own canid, initiate enumeration as automatic conflict resolution (Thanks to Bob V for this idea)
714 // we know enumerationInProgress = FALSE here
715 doEnum(FALSE);
716 enumerationStartTime.Val = tickGet(); // Start hold off time for self enumeration - start after 200ms delay
717 }
718
719 // Check for RTR - self enumeration request from another module
720
721 if (ptr->buffer[dlc] & 0x40 ) // RTR bit set?
722 {
723 TXB2CONbits.TXREQ = 1; // Send enumeration response (zero payload frame preloaded in TXB2)
724 enumerationStartTime.Val = tickGet(); // re-Start hold off time for self enumeration
725 }
726 else
727 {
728 msgFound = ptr->buffer[dlc] & 0x0F; // Check not zero payload
729 if (ptr->buffer[dlc] > 8)

738// Process transmit error interrupt
739
740void canTxError( void )
741{
742 if (TXB0CONbits.TXLARB) { // lost arbitration
743 if (larbRetryCount == 0) { // already tried higher priority
744 canTransmitFailed = TRUE;
745 canTransmitTimeout.Val = 0;
746
747 TXB0CONbits.TXREQ = 0;
748 larbCount++;
749 }
750 else if ( --larbRetryCount == 0) { // Allow tries at lower level priority first
751 TXB0CONbits.TXREQ = 0;
752 TXB0SIDH &= 0b00111111; // change to high priority ?? check priority bits usage
753 TXB0CONbits.TXREQ = 1; // try again
754 }
755 }
756 if (TXB0CONbits.TXERR) { // bus error
757 canTransmitFailed = TRUE;
758 canTransmitTimeout.Val = 0;
759 TXB0CONbits.TXREQ = 0;
760 txErrCount++;
761 }
762
763 if (canTransmitFailed)
764 checkTxFifo(); // Check to see if more to try and send
765
766 ERRIF = 0;
767}

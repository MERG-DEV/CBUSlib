125/**
126 * Check for FLiM button pressed and control entry into FLiM setup mode.
127 * The definition of FLiM_SW in hwsettings.h sets the I/O pin for the FLiM switch.
128 * A module that uses a mechanism for FLiM entry other than a simple push button
129 * can define FLiM_SW as a macro or procedure call as required.
130 */
131void FLiMSWCheck( void )
132{
133
134 switch (flimState)
135 {
136 case fsFLiM:
137 case fsSLiM:
138 if (!FLiM_SW)
139 {
140 prevFlimState = flimState;
141 flimState = fsPressed;
142
143 cbusMsg[d1] = (nodeID >>8)&0xff;
144 cbusMsg[d2] = nodeID & 0xff;
145 switchTime.Val = tickGet();
146 }
147 break;
148
149 case fsPressed:
150 if (!FLiM_SW)
151 {
152 if (tickTimeSince(switchTime) > FLiM_HOLD_TIME)
153 {
154 flimState = fsFlashing; // Held long enough so start flashing LED
155 startFLiMFlash(FALSE); // slow flash for FLiM setup
156 }
157 }
158 else if ((prevFlimState == fsFLiM) && (tickTimeSince(switchTime) > FLiM_DEBOUNCE_TIME)) // short press and release in FLiM
159 {
160 flimState = fsFLiMSetup;
161 startFLiMFlash(FALSE); // slow flash for FLiM setup
162 requestNodeNumber();
163 }
164 else
165 {
166 flimState = prevFlimState; // But not held down long enough, revert to previous state
167 }
168 break;
169
170 case fsFlashing:
171 if (FLiM_SW) // Button released
172 {
173 if (prevFlimState == fsSLiM)
174 {
175 flimState = fsFLiMSetup;
176 requestNodeNumber();
177
178 }
179 else
180 {
181 flimState = fsSLiM;
182 SLiMRevert();
183 setSLiMLed();
184 nodeID = 0;
185 SaveNodeDetails(nodeID, fsSLiM);
186 }
187 }
188#ifdef TEST_MODE
189 else if (tickTimeSince(switchTime) > SET_TEST_MODE_TIME)
190 {
191 flimState = fsPressedTest;
192 switchTime.Val = tickGet();
193 startFLiMFlash(TRUE); // Fast flash for test mode
194
195 }
196#endif // TEST_MODE
197 break;
198
199 case fsFLiMSetup: // Button pressed whilst in setup mode
200 if (!FLiM_SW)
201 {
202 flimState = fsPressedSetup; // Wait for debounce
203 switchTime.Val = tickGet();
204 }
205 break;
206
207 case fsPressedSetup: // Button was pressed whilst in setup mode
208 if (FLiM_SW)
209 {
210 if ((tickTimeSince(switchTime) > FLiM_DEBOUNCE_TIME)) // button released after debounce time
211 {
212 flimState = prevFlimState;
213 setStatusLed(flimState == fsFLiM);
214 }
215 else
216 flimState = fsFLiMSetup;
217 }
218 break;
219#ifdef TEST_MODE
220 case fsTestMode: // Button was pressed whilst in test mode
221 if (!FLiM_SW)
222 {
223 flimState = fsPressedTest;
224 switchTime.Val = tickGet();
225 }
226 break;
227
228 case fsPressedTest: // Button was pressed whilst in test mode
229 if (FLiM_SW) // button now released
230 if ((tickTimeSince(switchTime) > FLiM_HOLD_TIME)) // button released after full FLiM switch time
231 {
232 flimState = prevFlimState;
233 setStatusLed(flimState == fsFLiM);
234 }
235 else if ((tickTimeSince(switchTime) > NEXT_TEST_TIME)) // button released after time for next test selection
236 flimState = fsNextTest;
237
238 else if ((tickTimeSince(switchTime) > FLiM_DEBOUNCE_TIME)) // button released after debounce time
239 flimState = fsTestInput; // An input to the current test mode (eg: repeat test)
240 else
241 flimState = fsTestMode; // Button not held down long enough to do anything, so just carry on
242 break;
243 /*
244 case fsFLiMRelease:
245 case fsSetupDone:
246 case fsFLiMLearn:
247 case fsPressedFLiM:
248 case fsNextTest:
249 case fsTestInput:
250 default:
251 // Should never get here... but just in case
252 flimState = fsSLiM;
253 SLiMRevert();
254 setSLiMLed();
255 SaveNodeDetails(nodeID, fsSLiM);
256 break;
257 */
258#endif //TEST_MODE
259 } // switch
260} // FLiMSWCheck

291/**
292 * Process CBUS opcodes. Anything module explicit will already have been dealt
293 * with in the application code before calling here.
294 *
295 * @param msg the CBUS message
296 * @return true if the message was processed
297 */
298BOOL parseCBUSMsg(BYTE *msg) // Process the incoming message
299
300{
301 // check this is an EVENT
302 if (((msg[d0] & EVENT_SET_MASK) == EVENT_SET_MASK) && ((~msg[d0] & EVENT_CLR_MASK)== EVENT_CLR_MASK))
303 {
304 // it is an event pass to the module's event processing
305 return parseCbusEvent(msg);
306 }
307 // FLiM processing doesn't process Events
308 // Process the incoming (non-event) message
309 return( parseFLiMCmd(msg));
310}

313/**
314 * Process CBUS opcode for FLiM. Called after any module specific CBUS opcodes have
315 * been dealt with.
316 *
317 * @param rx_ptr the received CBUS message
318 * @return true if the message was processed
319 */
320BOOL parseFLiMCmd(BYTE *rx_ptr)
321{
322#ifdef __XC8__
323 BOOL cmdProcessed = FALSE;
324#else
325 overlay BOOL cmdProcessed = FALSE;
326#endif
327
328
329 if (flimState == fsFLiMLearn)
330 {
331 cmdProcessed = TRUE;
332
333 switch(rx_ptr[d0])
334 {
335 case OPC_NNLRN: // If in learn mode and receive a learn message for another node, must exit learn mode
336 if (thisNN(rx_ptr))
337 break; // Ignore if for us (already in learn) but drop through to exit learn mode if not addressed to us
338 /* fall through */
339 case OPC_NNULN:
340 // Release node from learn mode
341 flimState = fsFLiM;
342 break;
343
344 case OPC_NNCLR:
345 // Clear all events
346 doNnclr();
347 break;
348
349 case OPC_EVULN:
350 // Unlearn event
351 doEvuln( (((WORD)rx_ptr[d1]) << 8) + rx_ptr[d2], (((WORD)rx_ptr[d3]) << 8) + rx_ptr[d4]);
352 break;
353
354 case OPC_EVLRN:
355 // Teach event whilst in learn mode
356 doEvlrn( (((WORD)rx_ptr[d1]) << 8) + rx_ptr[d2] , (((WORD)rx_ptr[d3]) << 8) + rx_ptr[d4], rx_ptr[d5], rx_ptr[d6]);
357 break;
358
359 case OPC_EVLRNI:
360 // Teach event whilst in learn mode with event index
361 doEvlrn((((WORD)rx_ptr[d1]) << 8) + rx_ptr[d2] , (((WORD)rx_ptr[d3]) << 8) + rx_ptr[d4], rx_ptr[d6], rx_ptr[d7]); // Current implementation does not use index to save learnt event
362 break;
363
364 case OPC_REQEV:
365 // Read event variable by event id
366 doReqev((((WORD)rx_ptr[d1]) << 8) + rx_ptr[d2] , (((WORD)rx_ptr[d3]) << 8) + rx_ptr[d4], rx_ptr[d5]);
367 break;
368
369 default:
370 cmdProcessed = FALSE;
371 break;
372 }
373 } // in learn mode
374
375 if (!cmdProcessed && thisNN(rx_ptr))
376 { // process commands specifically addressed to us
377 cmdProcessed = TRUE;
378
379 switch(rx_ptr[d0])
380 {
381 case OPC_RQNPN:
382 // Read one node parameter by index
383 doRqnpn(rx_ptr[d3]);
384 break;
385
386 case OPC_NNLRN:
387 // Put node into learn mode
388 if (flimState == fsFLiM)
389 flimState = fsFLiMLearn;
390 break;
391
392 case OPC_NNEVN:
393 // Read available event slots
394 doNnevn();
395 break;
396
397 case OPC_NERD:
398 // Read all stored events
399 doNerd();
400 break;
401
402 case OPC_NENRD:
403 // Read a single stored event by index
404 doNenrd(rx_ptr[d3]);
405 break;
406
407 case OPC_RQEVN:
408 // Read number of stored events
409 doRqevn();
410 break;
411
412 case OPC_NVRD:
413 // Read value of a node variable
414 doNvrd(rx_ptr[d3]);
415 break;
416
417 case OPC_NVSET:
418 // Set a node variable
419 doNvset(rx_ptr[d3], rx_ptr[d4]);
420 break;
421
422 case OPC_REVAL:
423 // Read event variable by index
424 doReval(rx_ptr[d3], rx_ptr[d4]);
425 break;
426
427#ifdef BOOTLOADER_PRESENT
428 case OPC_BOOT:
429 // Enter bootloader mode
430 ee_write((WORD)EE_BOOT_FLAG, 0xFF);
431 Reset();
432 break;
433#endif
434
435 case OPC_CANID:
436 if (! setNewCanId(rx_ptr[d3])) {
437 doError(CMDERR_INVALID_EVENT); // seems a strange error code but that's what the spec says...
438 }
439 break;
440
441 case OPC_ENUM:
442 doEnum(TRUE);
443 break;
444
445#ifdef AREQ_SUPPORT
446 case OPC_AREQ:
447 // Illicit a response indicating last long event state
448 // The NN supplied is actually the Long event's NN instead of the addressed module
449 // but these should be the same for the producer of the event.
450 doAreq((((WORD)rx_ptr[d1]) << 8) + rx_ptr[d2] , (((WORD)rx_ptr[d3]) << 8) + rx_ptr[d4]);
451 cmdProcessed = TRUE;
452 break;
453
454 case OPC_ASRQ:
455 // Illicit a response indicating last short event state
456 // The NN supplied is actually the Long event's NN instead of the addressed module
457 // but these should be the same for the producer of the event.
458 doAreq(0 , (((WORD)rx_ptr[d3]) << 8) + rx_ptr[d4]);
459 cmdProcessed = TRUE;
460 break;
461#endif
462 default:
463 cmdProcessed = FALSE;
464 break;
465 }
466 } // this NN
467
468 if (!cmdProcessed)
469 { // Process any command not sent specifically to us that still needs action
470 switch (rx_ptr[d0]) {
471 case OPC_QNN:
472 QNNrespond(); // Respond to node query // ??? update to do new spec response agreed
473 cmdProcessed = TRUE;
474 break;
475 }
476 }
477
478 // In setup mode, also check for FLiM commands not addressed to
479 // any particular node
480
481 if ((!cmdProcessed) && (flimState == fsFLiMSetup))
482 {
483 cmdProcessed = TRUE;
484
485 switch(rx_ptr[d0])
486 {
487 case OPC_RQNP:
488 // Read node parameters
489 doRqnp();
490 break;
491
492 case OPC_RQMN:
493 // Read module type name
494 doRqmn();
495 break;
496
497 case OPC_SNN:
498 // Set node number
499 doSnn(rx_ptr);
500 break;
501
502
503 default:
504 cmdProcessed = FALSE;
505 break;
506 }
507 }
508
509 return( cmdProcessed );
510} // parse_FLiM_cmd

515/**
516 * return the current parameter flags
517 * @return parameter flags
518 */
519BYTE getParFlags() {
520 FLiMprmptr paramptr = (FLiMprmptr)&FLiMparams;
521 paramptr->params.module_flags;
522 if (flimState == fsFLiMLearn) {
523 return(PF_LRN | PF_FLiM) | (paramptr->params.module_flags);
524 }
525 if (flimState == fsFLiM) {
526 return(PF_FLiM)|(paramptr->params.module_flags);
527 }
528 return 0;
529}

548/**
549 * Read one node parameter by index.
550 */
551void doRqnpn(BYTE idx)
552{
553 FLiMprmptr paramptr;
554
555 paramptr = (FLiMprmptr)&FLiMparams;
556
557 if (idx <= FCUparams.parameter_count)
558 {
559 cbusMsg[d0] = OPC_PARAN;
560 cbusMsg[d3] = idx;
561
562 if (idx == 0)
563 cbusMsg[d4] = FCUparams.parameter_count;
564 else if ((idx >= PAR_CPUMID) && (idx < PAR_CPUMAN) )
565 cbusMsg[d4] = readCPUType() >> (( idx - PAR_CPUMID )*8);
566 else
567 cbusMsg[d4] = paramptr->bytes[idx-1];
568
569 if ((idx == PAR_FLAGS) && (flimState == fsFLiM))
570 cbusMsg[d4] |= PF_FLiM;
571
572 cbusSendMsgNN(ALL_CBUS, -1, cbusMsg);
573 }
574 else
575 {
576 doError(CMDERR_INV_PARAM_IDX);
577 }
578} // doRqnpn

581/**
582 * Read a node variable.
583 * Could just access the nodeVarTable directly but doing via the flash routines means the
584 * nearby values are cached.
585 * @param NVindex the index of the Node Variable
586 */
587void doNvrd(BYTE NVindex)
588{
589 // check the bounds of NVindex. It starts at 1
590 if ((NVindex == 0) || (NVindex >= NV_NUM))
591 {
592 doError(CMDERR_INV_NV_IDX);
593 }
594 else
595 {
596 WORD flashIndex;
597 flashIndex = AT_NV;
598 flashIndex += NVindex;
599
600 // Get NV index and send response with value of NV
601 cbusMsg[d0] = OPC_NVANS;
602 cbusMsg[d3] = NVindex;
603 cbusMsg[d4] = readFlashBlock(flashIndex);
604 cbusSendMsgMyNN( 0, cbusMsg );
605 }
606} // doNvrd

609/**
610 * Set a node variable.
611 * @param NVindex the index of the NV to be written
612 * @param NVvalue the new NV value
613 */
614void doNvset(BYTE NVindex, BYTE NVvalue)
615{
616 // check the bounds of NVindex. It starts at 1
617 if ((NVindex == 0) || (NVindex >= NV_NUM))
618 {
619 doError(CMDERR_INV_NV_IDX);
620 }
621 else
622 {
623 WORD flashIndex;
624 BYTE oldValue;
625
626 // *NVPtr[--NVindex] = NVvalue; // Set value of node variable (NV counts from 1 in opcode, adjust index to count from zero)
627
628 flashIndex = AT_NV;
629 flashIndex += NVindex;
630
631 oldValue = NvBytePtr[NVindex];
632 if (validateNV(NVindex, oldValue, NVvalue))
633 {
634 writeFlashByte((BYTE *)flashIndex, NVvalue);
635#ifdef NV_CACHE
636 loadNvCache();
637#endif
638 actUponNVchange(NVindex, oldValue, NVvalue);
639 cbusSendOpcMyNN( 0, OPC_WRACK, cbusMsg);
640 }
641 else
642 {
643 doError(CMDERR_INV_NV_VALUE);
644 }
645 }
646} // doNvset

650/**
651 * Read node parameters in setup mode.
652 * Returns the first set of parameters that will fit in a CAN message.
653 * Additional parameters added since this opcode was defined can only be read using RQNPN.
654 */
655void doRqnp(void)
656{
657 FLiMprmptr paramptr;
658 BYTE copyCounter;
659
660 paramptr = (FLiMprmptr)&FLiMparams;
661
662 cbusMsg[d0] = OPC_PARAMS;
663
664 for (copyCounter = d1; copyCounter <= d7; copyCounter++)
665 {
666 cbusMsg[copyCounter] = paramptr->bytes[copyCounter-d1];
667 }
668
669 // update the dynamic flags
670 cbusMsg[d1+PAR_FLAGS-1] = getParFlags();
671
672 cbusSendMsg(0, cbusMsg);
673
674} // doRqnp

676/** Read node parameters in setup mode.
677 * Returns the first set of parameters that will fit in a CAN message.
678 * Additional parameters added since this opcode was defined can only be read using RQNPN.
679 */
680void doRqmn(void)
681{
682 BYTE copyCounter, padCounter;
683#ifdef __XC8__
684 char* namptr;
685#else
686 rom char* namptr;
687#endif
688 DWORD namadr;
689
690
691 namadr = FCUparams.module_type_name;
692#ifdef __XC8__
693 namptr = (char*)namadr;
694#else
695 namptr = (rom char*)namadr;
696#endif
697
698
699
700 cbusMsg[d0] = OPC_NAME;
701 // This MUST be 7 characters.
702 for (copyCounter = 0; copyCounter < 7; copyCounter++ )
703 cbusMsg[copyCounter+d1] = *namptr++;
704 // The source module_type_name string is now padded with spaces so no need to do it here.
705
706 cbusSendMsg( 0, cbusMsg );
707
708} // doRqmn

710/**
711 * Set node number while in setup mode.
712 * @param rx_ptr the CBUS message
713 */
714void doSnn( BYTE *rx_ptr )
715{
716
717 // Get new node number for FLiM
718 nodeID = rx_ptr[d1];
719 nodeID <<= 8;
720 nodeID += rx_ptr[d2];
721 flimState = fsFLiM;
722
723 // Store new node id and mode to nonvol EEPROM
724
725 SaveNodeDetails(nodeID, fsFLiM);
726
727 // Acknowledge new node id
728
729 cbusSendOpcMyNN( 0, OPC_NNACK, cbusMsg );
730 setFLiMLed();
731
732} // doSnn

746/**
747 * Test to see if the CBUS message matches our NN.
748 * @param rx_ptr the CBUS message
749 * @return Trus if NN matches ours
750 */
751BOOL thisNN( BYTE *rx_ptr)
752{
753 if (((rx_ptr[d0] >> 5) >= 2) && (((WORD)(rx_ptr[d1])<<8) + rx_ptr[d2]) == nodeID)
754 return TRUE;
755 else
756 return FALSE;
757}
